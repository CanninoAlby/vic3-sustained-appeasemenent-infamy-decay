pp_infamy_gdp_one_per = {	
	subtract = {
		value = root.gdp
		multiply = 0.01		#1%
	}
	subtract = {
		value = root.infamy
		multiply = 0.001
	}
	max = {
        value = root.gdp
        multiply = 0.04		#gdp的4%
    }
}

		}
