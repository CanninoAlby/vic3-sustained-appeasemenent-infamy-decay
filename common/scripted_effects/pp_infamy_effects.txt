pp_calculate_target_gdp = {
	# We evaluate the script values directly which handles the GDP calculations natively,
	# avoiding nested scope errors in scripted effects.
	set_variable = { name = pp_locked_cost_1 value = pp_cost_tier_1 }
	set_variable = { name = pp_locked_cost_2 value = pp_cost_tier_2 }
	set_variable = { name = pp_locked_cost_3 value = pp_cost_tier_3 }
}

pp_process_weekly_appeasement = {
	# We only need to distribute money to the AI.
	# The Player's expense is handled automatically by the modifier_pp_appeasement_expense!
	
	if = {
		limit = { has_modifier = modifier_positive_propaganda.1 }
		set_variable = { name = temp_rate value = 0.001 }
	}
	else_if = {
		limit = { has_modifier = modifier_positive_propaganda.2 }
		set_variable = { name = temp_rate value = 0.003 }
	}
	else_if = {
		limit = { has_modifier = modifier_positive_propaganda.3 }
		set_variable = { name = temp_rate value = 0.006 }
	}
	else = {
		set_variable = { name = temp_rate value = 0 }
	}
	
	if = {
		limit = { var:temp_rate > 0 }
		every_country = {
			limit = {
				NOT = { this = root }
				country_rank >= rank_value:major_power
			}
			# Give money to AI based on their GDP and the active rate
			add_treasury = {
				value = gdp
				multiply = root.var:temp_rate
			}
		}
	}
}
